#Branch Not Equal Test:



		.data
string1		.asciiz	"Failed to branch if not equal.\n"		# declaration for string variable, 
string2		.asciiz	"Successfully Executed BNE.\n"		# declaration for string variable, 




		.text
main:	
li $v0, 4		# load appropriate system call code into register $v0;
			# code for printing string is 4



addi $t0, $zero, 5
addi $t1, $zero, 4
bne $t1, $t2, success

error:
la $a0, string1		# load address of string to be printed into $a0
syscall	

success:
la $a0, string2		# load address of string to be printed into $a0
syscall
